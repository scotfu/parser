 234 def remove_comments(tokens):

 237 def format_quoted_string(string):
 238     out =[]
 239     length = len(string)
 240     i = 0
 241     while i < length-1:
 242         if string[i] == '\\':
 243             i += 1
 244             if string[i] == 'n':
 245                 out.append('\12')
 246             elif string[i] == 'r':
 247                 out.append('\15')
 248             else:
 249                 out.append(string[i])
 250         else:
 251             out.append(string[i])
 252         i += 1
 253     return '""' + ''.join(out) + '"""'

 256 def run():
 257     file_name = 'test.cfg'
 258     error = ''
 259     if len(sys.argv) > 1:
 260         file_name = sys.argv[1]
 261     if not os.path.isfile(file_name):
 262         sys.stdout.write('ERR:F:\n')
 263         exit()
 264     data = open(file_name).read()
 265     try:
 266         tokenizer = Tokenizer(data)
 267         tokenizer.tokenize()
 268     except (FlexException, IllegalException, NullException):
 269         sys.stdout.write('ERR:L:%d\n'%tokenizer.lineno)
 270         exit()
 271     except IndexError:
 272         sys.stdout.write('ERR:L:%d\n'%tokenizer.lineno)
 273         exit()
 274     try:    
 275         p = Parser()
 276         tokens = remove_comments(tokenizer.tokens)
 277         last_line = tokens[-1].lineno
 278         p.parse(tokens)
 279     except PException,e:
 280         error = 'ERR:P:{0}\n'.format(e)#bugs?
 281     except IndexError:
 282         error = 'ERR:P:{0}\n'.format(last_line)
 283     finally:
 284         p.print_tree()
 285         sys.stdout.write(error)
 286 if __name__ == '__main__':
 287     run()
